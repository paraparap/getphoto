# install & start

$ sudo apt update && apt upgrade -y <br>

$ sudo apt install git<br>

$ sudo git clone https://github.com/paraparap/getphoto<br>

$ cd getphoto<br>

$ sudo chmod +x getphoto.sh<br>

$ sudo ./getphoto.sh<br>
